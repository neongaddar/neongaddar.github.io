Aiogram Premium
from aiogram import Bot, Dispatcher, Router from aiogram.types import Message from aiogram.filters import Command from neonpay import NeonPay bot = Bot("YOUR_BOT_TOKEN") dp = Dispatcher() router = Router() # Initialize NEONPAY neonpay = NeonPay.create_for_aiogram(bot) # Create premium stage premium_stage = neonpay.create_stage( title="Premium Access", description="Unlock all premium features", price=100, # 100 XTR logo_url="https://example.com/premium.png" ) @router.message(Command("premium")) async def buy_premium(message: Message): await neonpay.send_payment(message.chat.id, premium_stage) @neonpay.payment_handler(premium_stage) async def handle_premium(payment_result): user_id = payment_result.user_id # Grant premium access logic here await payment_result.message.answer( "🎉 Premium activated! Welcome to Premium!" ) dp.include_router(router)

Pyrogram Store
from pyrogram import Client, filters from pyrogram.types import Message, InlineKeyboardMarkup, InlineKeyboardButton from neonpay import NeonPay app = Client("my_bot", bot_token="YOUR_TOKEN") neonpay = NeonPay.create_for_pyrogram(app) # Create product stages products = { "ebook": neonpay.create_stage( title="Python E-Book", description="Complete Python programming guide", price=50 ), "course": neonpay.create_stage( title="Video Course", description="10-hour Python video course", price=200 ), "template": neonpay.create_stage( title="Bot Template", description="Ready-to-use Telegram bot template", price=75 ) } @app.on_message(filters.command("store")) async def show_store(client, message: Message): keyboard = InlineKeyboardMarkup([ [InlineKeyboardButton("📚 E-Book (50 XTR)", callback_data="buy_ebook")], [InlineKeyboardButton("🎥 Course (200 XTR)", callback_data="buy_course")], [InlineKeyboardButton("🤖 Template (75 XTR)", callback_data="buy_template")] ]) await message.reply("🛍️ Welcome to our store!", reply_markup=keyboard)
