Simple Premium Access
Basic premium subscription with Aiogram
Aiogram Premium
from aiogram import Bot, Dispatcher, Router from aiogram.types import Message from aiogram.filters import Command from neonpay import NeonPay bot = Bot("YOUR_BOT_TOKEN") dp = Dispatcher() router = Router() # Initialize NEONPAY neonpay = NeonPay.create_for_aiogram(bot) # Create premium stage premium_stage = neonpay.create_stage( title="Premium Access", description="Unlock all premium features", price=100, # 100 XTR logo_url="https://example.com/premium.png" ) @router.message(Command("premium")) async def buy_premium(message: Message): await neonpay.send_payment(message.chat.id, premium_stage) @neonpay.payment_handler(premium_stage) async def handle_premium(payment_result): user_id = payment_result.user_id # Grant premium access logic here await payment_result.message.answer( "üéâ Premium activated! Welcome to Premium!" ) dp.include_router(router)
Digital Store
Multiple products with Pyrogram
Pyrogram Store
from pyrogram import Client, filters from pyrogram.types import Message, InlineKeyboardMarkup, InlineKeyboardButton from neonpay import NeonPay app = Client("my_bot", bot_token="YOUR_TOKEN") neonpay = NeonPay.create_for_pyrogram(app) # Create product stages products = { "ebook": neonpay.create_stage( title="Python E-Book", description="Complete Python programming guide", price=50 ), "course": neonpay.create_stage( title="Video Course", description="10-hour Python video course", price=200 ), "template": neonpay.create_stage( title="Bot Template", description="Ready-to-use Telegram bot template", price=75 ) } @app.on_message(filters.command("store")) async def show_store(client, message: Message): keyboard = InlineKeyboardMarkup([ [InlineKeyboardButton("üìö E-Book (50 XTR)", callback_data="buy_ebook")], [InlineKeyboardButton("üé• Course (200 XTR)", callback_data="buy_course")], [InlineKeyboardButton("ü§ñ Template (75 XTR)", callback_data="buy_template")] ]) await message.reply("üõçÔ∏è Welcome to our store!", reply_markup=keyboard) @app.on_callback_query(filters.regex(r"buy_(\w+)")) async def handle_purchase(client, callback_query): product_key = callback_query.data.split("_")[1] stage = products[product_key] await neonpay.send_payment(callback_query.message.chat.id, stage) @neonpay.payment_handler(products["ebook"]) async def deliver_ebook(payment_result): await payment_result.message.answer_document( document="ebook.pdf", caption="üìö Here's your Python E-Book! Enjoy learning!" )
Subscription System
Monthly subscriptions with python-telegram-bot
PTB Subscription
from telegram.ext import Application, CommandHandler, CallbackQueryHandler from telegram import Update, InlineKeyboardMarkup, InlineKeyboardButton from neonpay import NeonPay import datetime application = Application.builder().token("YOUR_TOKEN").build() neonpay = NeonPay.create_for_ptb(application) # Subscription plans plans = { "monthly": neonpay.create_stage( title="Monthly Premium", description="Premium access for 1 month", price=300 ), "yearly": neonpay.create_stage( title="Yearly Premium", description="Premium access for 1 year (2 months free!)", price=3000 ) } async def subscribe_command(update: Update, context): keyboard = InlineKeyboardMarkup([ [InlineKeyboardButton("üìÖ Monthly (300 XTR)", callback_data="sub_monthly")], [InlineKeyboardButton("üìÜ Yearly (3000 XTR)", callback_data="sub_yearly")] ]) await update.message.reply_text( "Choose your subscription plan:", reply_markup=keyboard ) async def handle_subscription(update: Update, context): query = update.callback_query plan_type = query.data.split("_")[1] stage = plans[plan_type] await neonpay.send_payment(query.message.chat_id, stage) @neonpay.payment_handler(plans["monthly"]) async def activate_monthly(payment_result): user_id = payment_result.user_id expiry = datetime.datetime.now() + datetime.timedelta(days=30) # Save subscription to database await save_subscription(user_id, "monthly", expiry) await payment_result.message.reply_text( "‚úÖ Monthly subscription activated until " + expiry.strftime("%Y-%m-%d") ) application.add_handler(CommandHandler("subscribe", subscribe_command)) application.add_handler(CallbackQueryHandler(handle_subscription, pattern=r"sub_\w+"))
Donation Bot
Flexible donations with pyTelegramBotAPI
telebot Donations
import telebot from telebot.types import InlineKeyboardMarkup, InlineKeyboardButton from neonpay import NeonPay bot = telebot.TeleBot("YOUR_TOKEN") neonpay = NeonPay.create_for_telebot(bot) # Donation amounts donations = { "small": neonpay.create_stage( title="Small Donation", description="Support our project with a small donation", price=25 ), "medium": neonpay.create_stage( title="Medium Donation", description="Help us grow with a medium donation", price=100 ), "large": neonpay.create_stage( title="Large Donation", description="Make a big impact with a large donation", price=500 ) } @bot.message_handler(commands=['donate']) def donate_command(message): keyboard = InlineKeyboardMarkup() keyboard.row( InlineKeyboardButton("‚òï Small (25 XTR)", callback_data="donate_small"), InlineKeyboardButton("üçï Medium (100 XTR)", callback_data="donate_medium") ) keyboard.row( InlineKeyboardButton("üéÅ Large (500 XTR)", callback_data="donate_large") ) bot.reply_to(message, "üíù Thank you for considering a donation!", reply_markup=keyboard) @bot.callback_query_handler(func=lambda call: call.data.startswith('donate_')) def handle_donation(call): amount_type = call.data.split('_')[1] stage = donations[amount_type] neonpay.send_payment_sync(call.message.chat.id, stage) @neonpay.payment_handler(donations["small"]) async def thank_small_donor(payment_result): await payment_result.message.reply_text( "‚òï Thank you for your small donation! Every bit helps!" ) @neonpay.on_payment async def log_all_donations(payment_result): # Log all donations to database/file print(f"Donation received: {payment_result.amount} XTR from user {payment_result.user_id}") bot.infinity_polling()
